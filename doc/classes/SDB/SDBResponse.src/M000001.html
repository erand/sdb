    def to_s        
      result = ""
      result += "\tRequest Id: " + @request_id + "\n"         if @request_id
      result += "\tBox Usage: " + @box_usage + "\n"           if @box_usage
      result += "\tError Code: " + @error_code + "\n"         if @error_code
      result += "\tError Message: " + @error_message + "\n"   if @error_message
      
      return result
    end
